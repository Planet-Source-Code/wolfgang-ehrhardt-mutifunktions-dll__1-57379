Function Dialog.FontDlg

Öffnet eine Font-Auswahldialog und gibt den gewählten Font zurück

Deklaration

Public Function FontDlg(ByVal hWnd As Long, _

                                            FontName As String, _

                                            FontSize As Long, _

                                            FontColor As Long, _

                                            FontBold As Boolean, _

                                            FontItalic As Boolean, _

                                            FontUnderline As Boolean, _

                                            FontStrikethrough As Boolean) _

                                                As Boolean

Argumente

hWnd

Hwnd des Owner

FontName

FontSize

Wird zur Rückgabe benötigt

FontColor

Wird zur Rückgabe benötigt

FontBold

Wird zur Rückgabe benötigt

Wird zur Rückgabe benötigt

Wird zur Rückgabe benötigt

FontStrikethrough

Wird zur Rückgabe benötigt

Private Sub Form_Load()

    Dim Fn As String

    Dim Fb As Boolean, Fi As Boolean, Fu As Boolean, Fst As Boolean

    Dim Fs As Long, Fc As Long

    Dim Dlg As New WindowAPI.Dialog

 

    If Dlg.FontDlg(Me.hWnd, Fn, Fs, Fc, Fb, Fi, Fu, Fst) Then _

        MsgBox "FontName = " & Fn & vbCrLf & _

                       "FontSize = " & Fs & vbCrLf & _

                       "FontColor = " & Fc & vbCrLf & _

                       "FontBold = " & Fb & vbCrLf & _

                       "FontItalic = " & Fi & vbCrLf & _

                        "FontUnderline = " & Fu & vbCrLf & _

                        "FontStrikethrough = " & Fst

   

    End

   

End Sub
